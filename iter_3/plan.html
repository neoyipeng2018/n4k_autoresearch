iter_3 Quant Research Hardening Implementation Plan

Goal: upgrade iter_3/ from a compact ETF momentum autoresearch sandbox into a more credible, auditable quant research harness. The implementation must preserve the core rule: select formal candidates by validation CAGR without ruin; use locked OOS, DBMF excess, drawdown, robustness, and anti-overfit statistics as diagnostics only.

For implementers: this document is a plan, not an implementation. Execute it in small TDD steps. Keep source changes focused, run uv run pytest -q, uv run mypy ., and uv run ruff check . after each phase, and commit after each completed phase. Do not let any code path select on locked OOS.

Architecture overview

The current repo already has the right skeleton: fixed evaluator in prepare.py, editable strategy in train.py, rationale/ledger schema in ideas/schema.py, public data modules in datasets/, and analysis notebooks. The hardening work adds six new infrastructure layers while removing stale OOS-selection semantics.

Acceptance criteria

  • Objective consistency: a repository-wide search for locked out-of-sample CAGR, previous_best_oos_cagr, and prioritizes_oos finds no stale selection language except where explicitly described as diagnostic history.
  • Validation gate: candidate status is keep only if non-ruined, finite, positive validation CAGR, and validation CAGR improves the previous best. OOS and DBMF never affect keep/discard.
  • Robustness: a reproducible grid evaluates lookback, top-N, universe subset, rebalance day, score weighting, risk cap, and cash fallback; analysis reports robust regions and median/p10/p90 metrics, not only the top row.
  • Walk-forward: the evaluator supports multiple non-overlapping validation windows or rolling train/validation/OOS folds with embargo, and produces fold-level diagnostics.
  • Data quality: long missing stretches produce warnings or failures; ETF inception is explicitly represented instead of silently zero-filled; price coverage appears in a manifest.
  • Public text features: FOMC, Beige Book, FRED, and SEC ingestion write timestamped artifacts with source URLs, hashes, publication timestamps, and coverage tests. Features are lagged before joining to prices.
  • Artifacts: every formal trial has a directory containing strategy params, code hash, data manifest, run log, result row, and environment metadata.
  • Anti-overfit reporting: analysis artifacts show trial count, family count, best-vs-median validation performance, and an explicit selection-bias warning.

Task 2.2 — Make strategy selector factory

The grid should not rewrite train.py for every row. Use a pure selector factory.

def make_momentum_selector(params: StrategyParams) -> WeightSelector:
    def lookback_return(series: pd.Series) -> float | None:
        clean = series.dropna()
        if len(clean) <= params.lookback_days:
            return None
        start = float(clean.iloc[-params.lookback_days - 1])
        end = float(clean.iloc[-1])
        if start <= 0.0 or end <= 0.0:
            return None
        return end / start - 1.0

    def selector(day: pd.Timestamp, prices: pd.DataFrame) -> dict[str, float]:
        del day
        scores: list[tuple[float, str]] = []
        for symbol in params.symbols:
            if symbol == params.cash_symbol or symbol not in prices.columns:
                continue
            score = lookback_return(prices[symbol])
            if score is not None:
                scores.append((score, symbol))
        if not scores:
            return {params.cash_symbol: 1.0}
        selected = sorted(scores, reverse=True)[: params.top_n]
        if selected[0][0] <= params.min_momentum:
            return {params.cash_symbol: 1.0}
        if params.score_weighted:
            positive = [(max(score, 0.0), symbol) for score, symbol in selected]
            total = sum(score for score, _symbol in positive)
            if total <= 0.0:
                return {params.cash_symbol: 1.0}
            return {symbol: params.risk_cap * score / total for score, symbol in positive}
        weight = params.risk_cap / len(selected)
        weights = {symbol: weight for _score, symbol in selected}
        if params.risk_cap < 1.0:
            weights[params.cash_symbol] = 1.0 - params.risk_cap
        return weights
    return selector

def run_robustness_grid(params_grid: Iterable[StrategyParams]) -> pd.DataFrame:
    rows: list[dict[str, object]] = []
    best_validation = 0.0
    for params in params_grid:
        selector = make_momentum_selector(params)
        result = evaluate_oos(selector, symbols=list(params.symbols), previous_best_validation_cagr=best_validation)

            "oos_cagr": result.oos_cagr,
            "benchmark_oos_cagr": result.benchmark_oos_cagr,
            "excess_oos_cagr_vs_dbmf": result.excess_oos_cagr_vs_dbmf,

        })

def summarize_robust_regions(frame: pd.DataFrame, group_cols: list[str]) -> pd.DataFrame:
    grouped = frame.loc[~frame["ruined"]].groupby(group_cols, dropna=False)
    return grouped.agg(
        rows=("validation_cagr", "size"),
        median_validation_cagr=("validation_cagr", "median"),
        p10_validation_cagr=("validation_cagr", lambda s: s.quantile(0.10)),
        p90_validation_cagr=("validation_cagr", lambda s: s.quantile(0.90)),
        median_oos_cagr=("oos_cagr", "median"),
        ruin_rate=("ruined", "mean"),
    ).reset_index().sort_values(["median_validation_cagr", "p10_validation_cagr"], ascending=False)

# research/walk_forward.py
from __future__ import annotations

from dataclasses import dataclass
import pandas as pd
from prepare import EvaluationSplit

@dataclass(frozen=True)
class WalkForwardFold:
    name: str
    train: EvaluationSplit
    validation: EvaluationSplit
    oos: EvaluationSplit
    embargo_days: int


def make_rolling_folds(
    start_year: int,
    end_year: int,
    train_years: int = 3,
    validation_years: int = 1,
    oos_years: int = 1,
    embargo_days: int = 21,
) -> tuple[WalkForwardFold, ...]:
    folds: list[WalkForwardFold] = []
    year = start_year
    while year + train_years + validation_years + oos_years - 1 <= end_year:
        train_start = f"{year}-01-01"
        train_end = f"{year + train_years - 1}-12-31"
        validation_start = (pd.Timestamp(train_end) + pd.Timedelta(days=embargo_days + 1)).date().isoformat()
        validation_end_year = year + train_years + validation_years - 1
        validation_end = f"{validation_end_year}-12-31"
        oos_start = (pd.Timestamp(validation_end) + pd.Timedelta(days=embargo_days + 1)).date().isoformat()
        oos_end_year = validation_end_year + oos_years
        oos_end = f"{oos_end_year}-12-31"
        folds.append(WalkForwardFold(
            name=f"wf_{year}_{oos_end_year}",
            train=EvaluationSplit("train", train_start, train_end, "train"),
            validation=EvaluationSplit("validation", validation_start, validation_end, "validation"),
            oos=EvaluationSplit("oos", oos_start, oos_end, "oos"),
            embargo_days=embargo_days,
        ))
        year += 1
    return tuple(folds)

# datasets/quality.py
from __future__ import annotations

from dataclasses import dataclass, asdict
import pandas as pd

@dataclass(frozen=True)
class PriceCoverageIssue:
    symbol: str
    severity: str
    message: str
    first_date: str | None = None
    last_date: str | None = None
    missing_count: int = 0
    max_missing_streak: int = 0


def max_consecutive_missing(series: pd.Series) -> int:
    longest = current = 0
    for is_missing in series.isna().to_list():
        current = current + 1 if is_missing else 0
        longest = max(longest, current)
    return longest


def inspect_price_coverage(
    prices: pd.DataFrame,
    max_missing_streak: int = 5,
    min_coverage_ratio: float = 0.95,
) -> list[PriceCoverageIssue]:
    issues: list[PriceCoverageIssue] = []
    for symbol in prices.columns:
        series = prices[symbol]
        non_missing = series.dropna()
        if non_missing.empty:
            issues.append(PriceCoverageIssue(symbol, "error", "no usable prices"))
            continue
        missing_count = int(series.isna().sum())
        coverage = 1.0 - missing_count / max(len(series), 1)
        streak = max_consecutive_missing(series)
        if streak > max_missing_streak:
            issues.append(PriceCoverageIssue(symbol, "error", "missing streak exceeds limit", str(non_missing.index[0].date()), str(non_missing.index[-1].date()), missing_count, streak))
        elif coverage < min_coverage_ratio:
            issues.append(PriceCoverageIssue(symbol, "warning", "coverage below threshold", str(non_missing.index[0].date()), str(non_missing.index[-1].date()), missing_count, streak))
    return issues

def fetch_text(url: str, timeout: int = 30) -> str:
    response = requests.get(url, headers={"User-Agent": "macro-autoresearch/0.1"}, timeout=timeout)

def write_fomc_events(documents: list[FomcDocument], cache_dir: Path) -> Path:
    rows: list[dict[str, str]] = []
    for doc in documents:
        text = fetch_text(doc.url)
        digest = sha256(text.encode("utf-8")).hexdigest()
        # Parse publication date from URL/title where available; require manual fallback if missing.
        published_at = infer_fomc_published_at(doc.url, doc.title)
        rows.append({
            "published_at": published_at,

            "title": doc.title,
            "url": doc.url,
            "text_sha256": digest,
            "text": text,
        })
    path = cache_dir / "fomc_events.tsv"
    pd.DataFrame(rows).to_csv(path, sep="\t", index=False)
    return path

from __future__ import annotations

from dataclasses import asdict, is_dataclass
from pathlib import Path
import json
import shutil


def write_trial_artifact(
    root: Path,
    idea_id: str,
    timestamp_utc: str,
    params: dict[str, object],
    result_row: dict[str, str],
    data_manifest: dict[str, object],
    run_log_path: Path,
    commit: str,
) -> Path:
    safe_time = timestamp_utc.replace(":", "").replace("Z", "Z")
    trial_dir = root / "artifacts" / "trials" / f"{safe_time}_{idea_id}"
    trial_dir.mkdir(parents=True, exist_ok=False)
    (trial_dir / "params.json").write_text(json.dumps(params, indent=2, sort_keys=True))
    (trial_dir / "result_row.json").write_text(json.dumps(result_row, indent=2, sort_keys=True))
    (trial_dir / "data_manifest.json").write_text(json.dumps(data_manifest, indent=2, sort_keys=True))
    (trial_dir / "commit.txt").write_text(commit + "\n")
    if run_log_path.exists():
        shutil.copy2(run_log_path, trial_dir / "run.log")
    return trial_dir

    entries = []
    for symbol in symbols:
        path = cache_path(symbol)
        if path.exists():
            entries.append({
                "symbol": symbol,

# research/overfit.py
from __future__ import annotations
import pandas as pd


def infer_family(idea_id: str) -> str:
    parts = idea_id.split("_")
    return "_".join(parts[:2]) if len(parts) >= 2 else idea_id


def selection_bias_summary(results: pd.DataFrame) -> dict[str, object]:
    clean = results.copy()
    clean["family"] = clean["idea_id"].map(infer_family)
    non_ruined = clean.loc[clean["ruined"].astype(str).str.lower() != "true"]
    best = non_ruined.sort_values("validation_cagr", ascending=False).iloc[0]
    return {
        "trial_count": int(len(clean)),
        "non_ruined_count": int(len(non_ruined)),
        "family_count": int(non_ruined["family"].nunique()),
        "best_idea_id": str(best["idea_id"]),
        "best_validation_cagr": float(best["validation_cagr"]),
        "median_validation_cagr": float(non_ruined["validation_cagr"].median()),
        "best_minus_median_validation_cagr": float(best["validation_cagr"] - non_ruined["validation_cagr"].median()),
        "warning": "Best validation row is selected from many related trials; treat CAGR as upward-biased until robust across folds and costs.",
    }
