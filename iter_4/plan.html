Implementation Plan: Make Public Text Data Operational in iter_4

This plan turns the current text-data promise into a usable, tested, point-in-time feature workflow for the autonomous macro autoresearch loop.

Goal: Make FOMC, Beige Book, SEC EDGAR, and FRED/ALFRED-style public data materially usable by autonomous experiments while preserving the project rule that strategy iterations edit only train.py.

Architecture: Build a fixed public-data layer under datasets/ that ingests artifacts into a consistent cache, constructs publication-lagged numeric feature frames, exposes a train-safe feature loader, records artifact provenance, and requires/validates actual non-price feature usage in the experiment loop.

Tech stack: Python 3.10, pandas, requests, pytest, mypy strict, ruff, TSV manifests, append-only results.tsv.

from __future__ import annotations

import time
import requests

HTTP_HEADERS = {
    "User-Agent": "n4k-autoresearch/0.1 research contact: public-web" 
}

def fetch_text_url(url: str, timeout: int = 30, delay_seconds: float = 0.2) -> str:
    response = requests.get(url, headers=HTTP_HEADERS, timeout=timeout)
    response.raise_for_status()
    time.sleep(delay_seconds)
    return response.text

from __future__ import annotations

from pathlib import Path
import pandas as pd

from datasets.cache import FOMC_DIR, BEIGE_BOOK_DIR, SEC_DIR, FRED_DIR

class MissingPublicDataError(RuntimeError):
    pass

def read_tsv(path: Path, required_columns: set[str]) -> pd.DataFrame:
    if not path.exists():
        raise MissingPublicDataError(f"Missing public data artifact: {path}")
    frame = pd.read_csv(path, sep="\t")
    missing = required_columns.difference(frame.columns)
    if missing:
        raise ValueError(f"{path} missing columns: {', '.join(sorted(missing))}")
    return frame

def load_fomc_events() -> pd.DataFrame:
    return read_tsv(FOMC_DIR / "fomc_events.tsv", {"published_at", "document_type", "url", "text"})

def load_beige_book_events() -> pd.DataFrame:
    return read_tsv(BEIGE_BOOK_DIR / "beige_book_events.tsv", {"published_at", "district", "url", "text"})

def load_sec_filings() -> pd.DataFrame:
    return read_tsv(SEC_DIR / "sec_filings.tsv", {"cik", "accession_number", "form_type", "accepted_at"})

def load_fred_observations() -> pd.DataFrame:
    return read_tsv(FRED_DIR / "fred_observations.tsv", {"series_id", "date", "realtime_start", "value"})

Task 3.2: Create datasets/public_features.py

This is the train-safe API. It returns only numeric features aligned to the supplied price index.

from __future__ import annotations

import pandas as pd

from datasets.loaders import load_fomc_events, load_beige_book_events, load_fred_observations
from datasets.text_features import (
    build_fomc_features,
    build_beige_book_features,
    align_event_features_to_prices,
)

def numeric_feature_columns(frame: pd.DataFrame) -> list[str]:
    return [name for name in frame.columns if name != "published_at" and pd.api.types.is_numeric_dtype(frame[name])]

def load_fomc_feature_frame(prices: pd.DataFrame) -> pd.DataFrame:
    events = load_fomc_events()
    features = build_fomc_features(events)
    numeric = features[["published_at", *numeric_feature_columns(features)]]
    aligned = align_event_features_to_prices(prices, numeric)
    return aligned.add_prefix("fomc_")

def load_public_feature_frame(prices: pd.DataFrame) -> pd.DataFrame:
    frames = [load_fomc_feature_frame(prices)]
    # Add Beige Book, SEC, and FRED frames as their builders land.
    return pd.concat(frames, axis=1).sort_index()

Task 4.1: Beige Book growth / inflation / labor pressure features

GROWTH_TERMS = frozenset({"growth", "expanded", "expansion", "strong", "solid", "robust"})
SLOWDOWN_TERMS = frozenset({"slowed", "slow", "weak", "declined", "softened", "contraction"})
INFLATION_TERMS = frozenset({"inflation", "prices", "costs", "wages", "pressure", "elevated"})
LABOR_TIGHT_TERMS = frozenset({"labor", "shortage", "hiring", "wage", "employment", "workers"})

def build_beige_book_features(events: pd.DataFrame) -> pd.DataFrame:
    required = {"published_at", "text"}
    missing = required.difference(events.columns)
    if missing:
        raise ValueError(f"Beige Book events missing required columns: {', '.join(sorted(missing))}")
    rows: list[dict[str, object]] = []
    for row in events.sort_values("published_at").itertuples(index=False):
        text = str(getattr(row, "text"))
        tokens = tokenize(text)
        rows.append({
            "published_at": pd.Timestamp(getattr(row, "published_at")),
            "token_count": len(tokens),
            "growth_minus_slowdown": lexicon_score(text, GROWTH_TERMS, SLOWDOWN_TERMS),
            "inflation_pressure": lexicon_score(text, INFLATION_TERMS, frozenset[str]()),
            "labor_tightness": lexicon_score(text, LABOR_TIGHT_TERMS, frozenset[str]()),
        })
    return pd.DataFrame(rows)

Task 4.2: SEC filing risk / uncertainty metadata features

Initial SEC feature should be based on filing metadata counts and optional text hashes, not full NLP over huge filings. Align by accepted_at.

def build_sec_metadata_features(filings: pd.DataFrame) -> pd.DataFrame:
    required = {"accepted_at", "form_type", "cik"}
    missing = required.difference(filings.columns)
    if missing:
        raise ValueError(f"SEC filings missing columns: {', '.join(sorted(missing))}")
    frame = filings.copy()
    frame["published_at"] = pd.to_datetime(frame["accepted_at"], utc=True)
    frame["is_10q"] = (frame["form_type"] == "10-Q").astype(float)
    frame["is_10k"] = (frame["form_type"] == "10-K").astype(float)
    daily = frame.groupby(frame["published_at"].dt.normalize()).agg(
        sec_filing_count=("cik", "count"), sec_10q_share=("is_10q", "mean"), sec_10k_share=("is_10k", "mean")
    ).reset_index().rename(columns={"published_at": "published_at"})
    return daily

Task 4.3: FRED numeric features

def build_fred_features(observations: pd.DataFrame) -> pd.DataFrame:
    frame = observations.copy()
    frame["published_at"] = pd.to_datetime(frame["realtime_start"], utc=True)
    frame["value"] = pd.to_numeric(frame["value"], errors="coerce")
    wide = frame.pivot_table(index="published_at", columns="series_id", values="value", aggfunc="last").sort_index()
    if {"DGS10", "DGS2"}.issubset(wide.columns):
        wide["yield_curve_10y_2y"] = wide["DGS10"] - wide["DGS2"]
    wide["published_at"] = wide.index
    return wide.reset_index(drop=True)

from datasets.loaders import MissingPublicDataError
from datasets.public_features import load_public_feature_frame

FEATURES: pd.DataFrame | None = None
FEATURES_AVAILABLE = False


def get_features(prices: pd.DataFrame) -> pd.DataFrame:
    global FEATURES, FEATURES_AVAILABLE
    if FEATURES is None:
        try:
            FEATURES = load_public_feature_frame(prices)
            FEATURES_AVAILABLE = True
        except MissingPublicDataError:
            FEATURES = pd.DataFrame(index=prices.index)
            FEATURES_AVAILABLE = False
    return FEATURES.reindex(prices.index)


def select_weights(day: pd.Timestamp, prices: pd.DataFrame) -> dict[str, float]:

        # Example: defensive tilt during hawkish policy tone.

    return price_momentum_weights(prices)
